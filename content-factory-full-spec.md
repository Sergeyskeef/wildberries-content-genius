from instagrapi import Client
from sqlalchemy.orm import Session
from database.models import ContentSource
from datetime import datetime

class InstagramParser:
    def __init__(self, username: str, password: str):
        self.client = Client()
        self.client.login(username, password)
    
    def parse_hashtag(self, hashtag: str, amount: int = 50) -> list[dict]:
        """–ü–∞—Ä—Å–∏—Ç Reels –∏ Carousel –ø–æ —Ö—ç—à—Ç–µ–≥—É"""
        medias = self.client.hashtag_medias_recent(hashtag, amount=amount)
        
        results = []
        for media in medias:
            data = {
                "url": f"https://instagram.com/p/{media.pk}/",
                "platform": "instagram",
                "caption": media.caption_text,
                "likes": media.like_count,
                "comments": media.comments_count,
                "views": media.play_count if media.media_type == 2 else None,  # 2 = VIDEO
                "author": media.user.username,
                "author_followers": media.user.follower_count,
                "media_type": "reel" if media.media_type == 2 else "carousel",
                "video_url": media.video_url if media.media_type == 2 else None,
                "thumbnail_url": media.thumbnail_url or media.get_first_carousel_media().thumbnail_url,
                "parsed_at": datetime.utcnow(),
            }
            results.append(data)
        
        return results
    
    def parse_accounts(self, account_list: list[str]) -> list[dict]:
        """–ü–∞—Ä—Å–∏—Ç –ø–æ—Å–ª–µ–¥–Ω–∏–µ –ø–æ—Å—Ç—ã —Å –∫–æ–Ω–∫—É—Ä–µ–Ω—Ç—Å–∫–∏—Ö –∞–∫–∫–∞—É–Ω—Ç–æ–≤"""
        results = []
        for account in account_list:
            try:
                user_id = self.client.user_id_from_username(account)
                medias = self.client.user_medias_paginated(user_id, 10)
                
                for media in medias:
                    if media.media_type == 2:  # —Ç–æ–ª—å–∫–æ –≤–∏–¥–µ–æ (Reels)
                        data = {
                            "url": f"https://instagram.com/p/{media.pk}/",
                            "platform": "instagram",
                            "caption": media.caption_text,
                            "likes": media.like_count,
                            "comments": media.comments_count,
                            "views": media.play_count,
                            "author": account,
                            "author_followers": media.user.follower_count,
                            "media_type": "reel",
                            "video_url": media.video_url,
                            "thumbnail_url": media.thumbnail_url,
                            "parsed_at": datetime.utcnow(),
                        }
                        results.append(data)
            except Exception as e:
                print(f"Error parsing {account}: {e}")
                continue
        
        return results
    
    def save_to_db(self, db: Session, parsed_data: list[dict]):
        """–°–æ—Ö—Ä–∞–Ω—è–µ—Ç –∫–æ–Ω—Ç–µ–Ω—Ç –≤ –ë–î"""
        for item in parsed_data:
            # –ü—Ä–æ–≤–µ—Ä–∏—Ç—å, –Ω–µ –¥—É–±–ª—å –ª–∏
            existing = db.query(ContentSource).filter_by(url=item["url"]).first()
            if existing:
                continue
            
            source = ContentSource(
                url=item["url"],
                platform=item["platform"],
                caption=item["caption"],
                metadata=item,  # JSON
                status="pending",  # –∂–¥–µ—Ç –∞–Ω–∞–ª–∏–∑–∞
            )
            db.add(source)
        
        db.commit()

from googleapiclient.discovery import build
from database.models import ContentSource
from sqlalchemy.orm import Session

class YouTubeParser:
    def __init__(self, api_key: str):
        self.youtube = build('youtube', 'v3', developerKey=api_key)
    
    def search_trending_videos(self, query: str, max_results: int = 50) -> list[dict]:
        """–ü–æ–∏—Å–∫ —Ç—Ä–µ–Ω–¥–æ–≤—ã—Ö –≤–∏–¥–µ–æ –ø–æ –∑–∞–ø—Ä–æ—Å—É"""
        request = self.youtube.search().list(
            part='snippet',
            q=query,
            type='video',
            maxResults=max_results,
            order='viewCount',  # –ø–æ –ø—Ä–æ—Å–º–æ—Ç—Ä–∞–º
            publishedAfter='2025-11-01T00:00:00Z',  # –∑–∞ –ø–æ—Å–ª–µ–¥–Ω–∏–π –º–µ—Å—è—Ü
            relevanceLanguage='ru',
        )
        response = request.execute()
        
        results = []
        for item in response.get('items', []):
            video_id = item['id']['videoId']
            
            # –ü–æ–ª—É—á–∏—Ç—å —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫—É –≤–∏–¥–µ–æ
            stats_request = self.youtube.videos().list(
                part='statistics,contentDetails',
                id=video_id
            )
            stats_response = stats_request.execute()
            stats = stats_response['items'][0]['statistics']
            
            data = {
                "url": f"https://youtube.com/watch?v={video_id}",
                "platform": "youtube",
                "title": item['snippet']['title'],
                "description": item['snippet']['description'],
                "author": item['snippet']['channelTitle'],
                "thumbnail_url": item['snippet']['thumbnails']['high']['url'],
                "views": int(stats.get('viewCount', 0)),
                "likes": int(stats.get('likeCount', 0)),
                "comments": int(stats.get('commentCount', 0)),
                "published_at": item['snippet']['publishedAt'],
                "parsed_at": datetime.utcnow(),
            }
            results.append(data)
        
        return results
```

import json
import textwrap

class CarouselGenerator:
    def __init__(self, theme: str = "dark"):
        self.theme = theme
        self.width = 1080
        self.height = 1350
        self.colors = self._load_theme()
        self.fonts = self._load_fonts()
    
    def _load_theme(self) -> dict:
        """–ó–∞–≥—Ä—É–∑–∏—Ç—å —Ü–≤–µ—Ç–æ–≤—É—é —Å—Ö–µ–º—É"""
        themes = {
            "dark": {
                "bg": "#0f172a",
                "text_primary": "#ffffff",
                "text_secondary": "#94a3b8",
                "accent": "#3b82f6",
                "border": "#1e293b",
            },
            "light": {
                "bg": "#ffffff",
                "text_primary": "#0f172a",
                "text_secondary": "#64748b",
                "accent": "#2563eb",
                "border": "#e2e8f0",
            },
        }
        return themes.get(self.theme, themes["dark"])
    
    def _load_fonts(self) -> dict:
        """–ó–∞–≥—Ä—É–∑–∏—Ç—å —à—Ä–∏—Ñ—Ç—ã"""
        return {
            "title": ImageFont.truetype("fonts/Roboto-Bold.ttf", 64),
            "headline": ImageFont.truetype("fonts/Roboto-Bold.ttf", 48),
            "body": ImageFont.truetype("fonts/Roboto-Regular.ttf", 32),
            "small": ImageFont.truetype("fonts/Roboto-Regular.ttf", 24),
        }
    
    def generate_slide(self, slide_data: dict) -> Image.Image:
        """–°–æ–∑–¥–∞—Ç—å –æ–¥–∏–Ω —Å–ª–∞–π–¥ (PNG)"""
        
        # –°–æ–∑–¥–∞—Ç—å –Ω–æ–≤–æ–µ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ
        img = Image.new('RGB', (self.width, self.height), self.colors["bg"])
        draw = ImageDraw.Draw(img)
        
        # –î–æ–±–∞–≤–∏—Ç—å –Ω–æ–º–µ—Ä —Å–ª–∞–π–¥–∞ –≤ —É–≥–ª—É
        slide_num = f"{slide_data['number']}/10"
        draw.text((30, 30), slide_num, fill=self.colors["text_secondary"], 
                  font=self.fonts["small"])
        
        # –î–æ–±–∞–≤–∏—Ç—å –∑–∞–≥–æ–ª–æ–≤–æ–∫ (headline)
        headline = slide_data.get("headline", "")
        wrapped_headline = textwrap.fill(headline, width=20)
        draw.text((60, 150), wrapped_headline, fill=self.colors["accent"],
                  font=self.fonts["headline"])
        
        # –î–æ–±–∞–≤–∏—Ç—å –æ—Å–Ω–æ–≤–Ω–æ–π —Ç–µ–∫—Å—Ç (body_text)
        body_text = slide_data.get("body_text", "")
        wrapped_body = textwrap.fill(body_text, width=30)
        draw.text((60, 450), wrapped_body, fill=self.colors["text_primary"],
                  font=self.fonts["body"])
        
        # –ï—Å–ª–∏ –µ—Å—Ç—å CTA, –¥–æ–±–∞–≤–∏—Ç—å –∫–Ω–æ–ø–∫—É
        if slide_data.get("cta"):
            draw.rectangle((60, 1200, 1020, 1280), 
                          outline=self.colors["accent"], width=3)
            draw.text((100, 1220), slide_data["cta"],
                     fill=self.colors["accent"], font=self.fonts["body"])
        
        return img
    
    def generate_carousel(self, carousel_plan: dict, output_dir: Path) -> Path:
        """–°–æ–∑–¥–∞—Ç—å –≤—Å—é –∫–∞—Ä—É—Å–µ–ª—å (–≤—Å–µ —Å–ª–∞–π–¥—ã + ZIP)"""
        
        carousel_id = carousel_plan.get("title").replace(" ", "_").lower()
        carousel_dir = output_dir / carousel_id
        carousel_dir.mkdir(parents=True, exist_ok=True)
        
        # –ì–µ–Ω–µ—Ä–∏—Ä–æ–≤–∞—Ç—å –∫–∞–∂–¥—ã–π —Å–ª–∞–π–¥
        for slide_data in carousel_plan["slides"]:
            img = self.generate_slide(slide_data)
            slide_filename = f"slide_{slide_data['number']:02d}_{slide_data['type']}.png"
            img.save(carousel_dir / slide_filename)
        
        # –°–æ—Ö—Ä–∞–Ω–∏—Ç—å –º–µ—Ç–∞–¥–∞–Ω–Ω—ã–µ
        metadata = {
            "title": carousel_plan["title"],
            "description": carousel_plan["description"],
            "hashtags": ["#–≤–∞–ª–±–µ—Ä—Ä–∏–∑", "#–º–∞—Ä–∫–µ—Ç–ø–ª–µ–π—Å", "#–ø—Ä–æ–¥–∞–∂–∏"],
            "created_at": datetime.utcnow().isoformat(),
            "total_slides": len(carousel_plan["slides"]),
        }
        
        with open(carousel_dir / "metadata.json", "w") as f:
            json.dump(metadata, f, ensure_ascii=False, indent=2)
        
        # –°–æ–∑–¥–∞—Ç—å ZIP
        import shutil
        zip_path = output_dir / f"{carousel_id}.zip"
        shutil.make_archive(str(zip_path.with_suffix("")), "zip", carousel_dir)
        
        return zip_path
```
